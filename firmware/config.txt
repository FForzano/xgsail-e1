# SailFrames E1 configuration — place on the SD card root as /config.txt.
# Fields not listed here fall back to firmware defaults (config.h's Config
# struct). GPS/IMU sample rates are firmware-baked (10 Hz), not configurable
# here.

# Boat identifier (shown on the TFT splash screen, used in filenames and
# as the ESP-NOW mesh sender_id). Not the device's upload identity — that's
# external_id (the WiFi MAC), established once at claim time (see below).
boat_id=E1

# XGSail backend base URL. All device-protocol calls (claim/confirm,
# session-uploads, health) are relative to this.
api_base_url=https://xgsail.example.com

# One-time claim code from the XGSail app (docs/device-protocol.md §2).
# Redeemed automatically the first time WiFi connects after boot, then
# ignored — the device is claimed from then on (see /device.txt). Leave
# blank if claiming via the `claim <CODE>` console command or the app's
# BLE relay instead.
claim_code=

# Up to 5 WiFi networks, tried in order until one connects. Optional: the
# device can also be claimed and uploaded to entirely over the app's BLE
# relay, with no WiFi configured at all.
wifi1_ssid=YourHomeNetwork
wifi1_pass=changeme

wifi2_ssid=YachtClubNetwork
wifi2_pass=changeme

# Calypso Mini wind sensor heading offset in degrees (mounting correction).
wind_offset=0

# Recording is manual (button press, console rec/stoprec, or BLE
# start-rec/stop-rec) — see firmware/README.md. start_speed_knots
# survives only as the "boat is moving" heuristic that aborts an
# in-progress upload cycle; stop_speed_knots/start_delay_sec/
# stop_delay_sec are unused, kept for older cards' compatibility.
start_speed_knots=1.5
stop_speed_knots=0.5
start_delay_sec=10
stop_delay_sec=180

# Deletes a session's files from the SD card right after they upload
# successfully (WiFi or BLE relay), instead of leaving them for the
# manual `cleanup`/`delup` console command. Also settable live over BLE
# (device_config, docs/ble-config.md).
auto_cleanup_uploads=true

# Fleet unit role: racing_boat, rc_signal, rc_pin, mark, committee_chase, spare.
unit_role=racing_boat

# RTK relative mode (docs/gnss-rtk.md). Off by default; enabling reconfigures
# the GNSS module and is a physical bring-up act, not a remote-safe toggle.
rtk_enabled=false
